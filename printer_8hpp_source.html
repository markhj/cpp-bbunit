197  static void printSummary(uint16_t passed, uint16_t failed, uint16_t errors) {

217  std::cout << " Assertions passed: " << std::to_string(passed);
218  } else {
219  std::cout << " " << pad("Total: " + std::to_string(passed + failed + errors), cellSize);
220  std::cout << " | " << pad("Passed: " + std::to_string(passed), cellSize);
221  std::cout << " | " << pad("Failed: " + std::to_string(failed), cellSize);
222  std::cout << " | " << pad("Errors: " + std::to_string(errors), cellSize);
223  }
