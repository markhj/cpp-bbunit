453  template<HasWhatMethod T>
454  ProvidesAssertions &assertException(const std::function<void()> &func,
455  const std::optional<std::string> &message = std::nullopt) noexcept(false) {
456  assert([&]() -> InternalResult {
457  try {
458  func();
459  } catch (T &e) {
460  return !message.has_value()
461  ? InternalResult{true}
462  : InternalResult{message.value() == e.what(), message.value(), e.what()};
463  } catch (std::exception &e) {
464  // When another type of exception is thrown, we will attempt to
465  // get a hold of its name, so we can populate a more useful message
466  // to the result feed.
467  return {false, getClassName(T()), getClassName(e)};
468  } catch (...) {
469  return {false, getClassName(T()), "<Unknown exception>"};
470  }
471  // No exception thrown
472  return {false, getClassName(T()), "<No exception thrown>"};
473  });
