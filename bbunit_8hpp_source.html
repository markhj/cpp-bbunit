464  template<HasWhatMethod T>
465  ProvidesAssertions &assertException(const std::function<void()> &func,
466  const std::optional<std::string> &message = std::nullopt) noexcept(false) {
467  assert([&]() -> InternalResult {
468  try {
469  func();
470  } catch (T &e) {
471  return !message.has_value()
472  ? InternalResult{true}
473  : InternalResult{message.value() == e.what(), message.value(), e.what()};

476  // get a hold of its name, so we can populate a more useful message

481  }
482  // No exception thrown
483  return {false, getClassName(T()), "<No exception thrown>"};
484  });
